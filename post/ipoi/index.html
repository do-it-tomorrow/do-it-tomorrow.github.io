Subtask 1

因为 n,m≤3000n,m\le 3000n,m≤3000，直接使用数组模拟此操作，时间复杂度为 O(nm)O(nm)O(nm) 可以通过。

#include<bits/stdc++.h>
#define int unsigned int
using namespace std;
const int N=3e5+1005;
int n,m,a[N];
int ask(int l,int r){
	int ans=0;
	for(int i=l;i<=r;i++){
		ans+=a[i];
	}
	return ans;
}
void change1(int l,int r,int v){
	for(int i=l;i<=r;i++){
		a[i]+=v;
	}
}
void change2(int l,int r,int v){
	for(int i=l;i<=r;i++){
		a[i]*=v;
	}
}
void change3(int l,int r,int v){
	for(int i=l;i<=r;i++){
		a[i]=v;
	}
}
void add_(int x,int k){
	for(int i=n;i>=x;i--){
		a[i+1]=a[i];
	}
	a[x]=k;
	n++;
}
void del(int x){
	for(int i=x;i<n;i++){
		a[i]=a[i+1];
	}
	n--;
}
signed main(){
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	cin>>m;
	for(int i=1,op,l,r,k,x;i<=m;i++){
		cin>>op;
		if(op==0){
			cin>>l>>r;
			l++,r++;
			cout<<ask(l,r)<<'\n';
		}
		if(op==1){
			cin>>l>>r>>k;
			l++,r++;
			change1(l,r,k);
		}
		if(op==2){
			cin>>l>>r>>k;
			l++,r++;
			change2(l,r,k);
		}
		if(op==3){
			cin>>l>>r>>k;
			l++,r++;
			change3(l,r,k);
		}
		if(op==4){
			cin>>x>>k;
			x++;
			add_(x,k);
		}
		if(op==5){
			cin>>x;
			x++;
			del(x);
		}
	}
	return 0;
}

Subtask 3

考虑使用分块进行维护。

对于完整的一块，修改时直接给对应的块进行懒惰标记。

在处理不完整的块时，将这一块进行下传即可。

时间复杂度为 O(mn)O(m\sqrt{n})O(mn​)。

#include<bits/stdc++.h>
#define int unsigned int
using namespace std;
const int N=550,inf=INT_MAX;
int n,t,len,cnt,a[N*N],sum[N],add[N],mul[N],change[N];
vector<int>s[N];
void pushdown(int x){
	if(change[x]!=inf){
		for(int i=0;i<s[x].size();i++){
			s[x][i]=change[x];
		}
	}
	if(mul[x]!=1){
		for(int i=0;i<s[x].size();i++){
			s[x][i]*=mul[x];
		}
	}
	if(add[x]!=0){
		for(int i=0;i<s[x].size();i++){
			s[x][i]+=add[x];
		}
	}
	mul[x]=1;
	add[x]=0;
	change[x]=inf;
}
int find(int &l){
	int x=1;
	while(l>s[x].size()){
		l-=s[x].size();
		x++;
	}
	return x;
}
int ask(int l,int r){
	int x=find(l),y=find(r);
	int ans=0;
	if(x==y){
		pushdown(x);
		for(int i=l-1;i<r;i++){
			ans+=s[x][i];
		}
		return ans;
	}
	for(int i=x+1;i<y;i++){
		ans+=sum[i];
	}
	pushdown(x);
	for(int i=l-1;i<s[x].size();i++){
		ans+=s[x][i];
	}
	pushdown(y);
	for(int i=0;i<r;i++){
		ans+=s[y][i];
	}
	return ans;
}
void change1(int l,int r,int v){
	int x=find(l),y=find(r);
	if(x==y){
		pushdown(x);
		for(int i=l-1;i<r;i++){
			s[x][i]+=v;
			sum[x]+=v;
		}
		return ;
	}
	for(int i=x+1;i<y;i++){
		sum[i]+=s[i].size()*v;
		add[i]+=v;
	}
	pushdown(x);
	for(int i=l-1;i<s[x].size();i++){
		s[x][i]+=v;
		sum[x]+=v;
	}
	pushdown(y);
	for(int i=0;i<r;i++){
		s[y][i]+=v;
		sum[y]+=v;
	}
}
void change2(int l,int r,int v){
	int x=find(l),y=find(r);
	if(x==y){
		pushdown(x);
		for(int i=l-1;i<r;i++){
			sum[x]+=s[x][i]*(v-1);
			s[x][i]*=v;
		}
		return ;
	}
	for(int i=x+1;i<y;i++){
		sum[i]*=v;
		add[i]*=v;
		mul[i]*=v;
	}
	pushdown(x);
	for(int i=l-1;i<s[x].size();i++){
		sum[x]+=s[x][i]*(v-1);
		s[x][i]*=v;
	}
	pushdown(y);
	for(int i=0;i<r;i++){
		sum[y]+=s[y][i]*(v-1);
		s[y][i]*=v;
	}
}
void change3(int l,int r,int v){
	int x=find(l),y=find(r);
	if(x==y){
		pushdown(x);
		for(int i=l-1;i<r;i++){
			sum[x]+=-s[x][i]+v;
			s[x][i]=v;
		}
		return ;
	}
	for(int i=x+1;i<y;i++){
		sum[i]=s[i].size()*v;
		add[i]=0,mul[i]=1,change[i]=v;
	}
	pushdown(x);
	for(int i=l-1;i<s[x].size();i++){
		sum[x]+=-s[x][i]+v;
		s[x][i]=v;
	}
	pushdown(y);
	for(int i=0;i<r;i++){
		sum[y]+=-s[y][i]+v;
		s[y][i]=v;
	}
}

	ios::sync_with_stdio(false);
	cin.tie(nullptr);

	for(int i=1;i<=n;i++){

Subtask 4

考虑使用分块进行维护。

对于插入，找到对应位置的块，直接暴力将 kkk 插入 xxx 的前面，对于删除操作同理，操作的时间复杂度为块长。

因为数据保证随机，所以块长应该为 n\sqrt{n}n​ 级别，总时间复杂度为 O(mn)O(m\sqrt{n})O(mn​)。

#include<bits/stdc++.h>
#define int unsigned int
using namespace std;
const int N=550,inf=INT_MAX;
int n,t,len,cnt,a[N*N],sum[N];
vector<int>s[N];
int find(int &l){
	int x=1;
	while(l>s[x].size()){
		l-=s[x].size();
		x++;
	}
	return x;
}
int ask(int l,int r){
	int x=find(l),y=find(r);
	int ans=0;
	if(x==y){
		for(int i=l-1;i<r;i++){
			ans+=s[x][i];
		}
		return ans;
	}
	for(int i=x+1;i<y;i++){
		ans+=sum[i];
	}
	for(int i=l-1;i<s[x].size();i++){
		ans+=s[x][i];
	}
	for(int i=0;i<r;i++){
		ans+=s[y][i];
	}
	return ans;
}
void add_(int x,int v){
	int k=find(x);
	s[k].insert(s[k].begin()+x-1,v);

void del(int x){

	sum[k]-=s[k][x-1];
	s[k].erase(s[k].begin()+x-1);

Subtask 7

因为 Subtask 6 的单次查询的时间复杂度为块长，所以在不随机的数据下，这样的程序可能会退化为 O(nm)O(nm)O(nm)。

所以，如果一块的长度大于 2n2\sqrt{n}2n​，那么就重新分块。重新分块的时间复杂度为 O(n)O(n)O(n)，所有的操作一共只会进行 m\sqrt{m}m​ 次合并，依然可以通过这个题目。
