з»ҷдҪ дёҖдёӘжңүеҗ‘еӣҫпјҢеҸҜд»Ҙд»ҳеҮәд»Јд»·иҺ·еҸ–дёҖдәӣжҢҮе®ҡзҡ„зӮ№гҖӮ
еңЁиҺ·еҸ–д№ӢеҗҺиҰҒжұӮиғҪд»ҘиҺ·еҸ–зҡ„зӮ№дёәеҮәеҸ‘зӮ№пјҢе°Ҷж•ҙдёӘеӣҫйғҪи®ҝй—®еҲ°пјҢжұӮжңҖе°Ҹзҡ„д»Јд»·гҖӮ

жҖқи·Ҝ

ж—ўз„¶йңҖиҰҒд»ӨжҖ»зҡ„д»Јд»·жңҖе°‘пјҢйӮЈд№ҲеҰӮжһңйҖҡиҝҮд№°дёҖдёӘзӮ№е°ұеҸҜд»Ҙи®ҝй—®еҲ°зҡ„жүҖжңүзӮ№пјҢиҮӘз„¶дјҡжҜ”д№°дёӨдёӘзӮ№зҡ„ж–№жЎҲжӣҙдјҳгҖӮ
дәҺжҳҜиҮӘз„¶зҡ„е°ұеҸҜд»ҘиҒ”жғіеҲ°еҸҜд»Ҙе°ҶеӣҫеҲ’еҲҶжҲҗеҫҲеӨҡдёӘејәиҝһйҖҡеӣҫпјҢеҸӘиҰҒеңЁиҝҷдёӘеӣҫдёӯжңүдёҖдёӘзӮ№и®ҝй—®еҲ°дәҶпјҢж•ҙдёӘејәиҝһйҖҡеӣҫе°ұиў«и®ҝй—®еҲ°дәҶгҖӮ

ж—ўз„¶иҰҒжұӮејәиҝһйҖҡеӣҫпјҢйӮЈд№Ҳе°ұиҮӘз„¶зҡ„йңҖиҰҒз”ЁеҲ°tarjanз®—жі•дәҶгҖӮ
еҶҚжұӮеҮәеӣҫеҶ…зҡ„жүҖз”ЁејәиҝһйҖҡеӣҫд№ӢеҗҺеҜ№е®ғд»¬иҝӣиЎҢзј©зӮ№пјҢиҝҷж ·е°ұеҸӘиҰҒиғҪи®ҝй—®иҝҷдёӘзӮ№е°ұеҸҜд»ҘдәҶгҖӮ

дҪҶжҳҜеӣ дёәеңЁзј©зӮ№д№ӢеҗҺе°ұдёҚиғҪз®ҖеҚ•зҡ„йҒҚеҺҶиҝҷдёӘејәиҝһйҖҡеӣҫпјҢжүҖд»ҘжҲ‘д»¬е°ұйңҖиҰҒе°ҶиҝҷдёӘејәиҝһйҖҡеӣҫдёӯд»Јд»·жңҖе°Ҹзҡ„дёҖдёӘзӮ№и®°еҪ•дёӢжқҘгҖӮ
иҝҷе°ұйңҖиҰҒдҪҝз”ЁдёҖдёӘиҫ…еҠ©ж•°з»„ min_cost жқҘи®°еҪ•гҖӮ
е…·дҪ“зҡ„е®һзҺ°д№ҹеҫҲз®ҖеҚ•пјҢеҸӘйңҖиҰҒеңЁе°Ҷж ҲеҶ…зҡ„е…ғзҙ еј№еҮәж—¶иҝӣиЎҢжҜ”иҫғе°ұеҸҜд»ҘдәҶгҖӮ

#include<bits/stdc++.h>
using namespace std;
int n,r,p,dfn[3200],low[3200],group[320![](https://do-it-tomorrow.github.io/post-images/1704207640718.jpg)0],in[3200],cost[3200],cnt,num,min_cost[3200],ans;
bool vis[3200];
vector<int> v[3200];
stack<int> s; 
void tarjan(int x){ //tarjanзҡ„жқҝеӯҗ+зј©зӮ№
	dfn[x]=low[x]=++num; 
	vis[x]=1;
	s.push(x);
	for(int i:v[x]){ //дҫқж¬ЎдҫҝеҲ©v[x]дёӯзҡ„е…ғзҙ иөӢеҖјз»ҷi
		if(!dfn[i]){
			tarjan(i);
			low[x]=min(low[x],low[i]);
		}else if(vis[i])
			low[x]=min(low[x],dfn[i]);
	}if(dfn[x]==low[x]){ //еҰӮжһңжңүзҺҜ
		group[x]=++cnt; 
		min_cost[cnt]=cost[x]; 
		vis[x]=0;
		while(s.top()!=x){ //ж ҲеҶ…жңүејәиҝһйҖҡеӣҫеҶ…зҡ„е…ғзҙ 
			group[s.top()]=cnt; 
			min_cost[cnt]=min(min_cost[cnt],cost[s.top()]);
			vis[s.top()]=0;
			s.pop();
		}s.pop(); 
	}return ;
}int find(int x){ //еҜ»жүҫиҝҷдёӘејәиҝһйҖҡеӣҫеҶ…жңҖе°Ҹзҡ„зӮ№
	for(int i=1;i<=n;i++)
		if(group[i]==x) 
			return i; 
	return n;
}int main(){
	memset(cost,0x3f,sizeof(cost));  //еҢәеҲҶжҳҜеҗҰеҸҜд»Ҙд№°
	cin>>n>>p;
	for(int i=1,a,b;i<=p;i++){
		cin>>a>>b;
		cost[a]=b;
	}cin>>r;
	for(int i=1,a,b;i<=r;i++){
		cin>>a>>b;
		v[a].push_back(b); //ж Үи®°еҸҜд»ҘйҖҡиҝҮaзӣҙжҺҘи®ҝй—®еҲ°b
	}for(int i=1;i<=n;i++){
		if(!dfn[i]&&cost[i]!=0x3f3f3f3f3f) //иҰҒеҲӨж–ӯжҳҜеҗҰеҸҜд»Ҙд№°
			tarjan(i);
	}for(int i=1;i<=n;i++){
		for(int j:v[i]){ //дҫқж¬ЎдҫҝеҲ©v[i]дёӯзҡ„е…ғзҙ иөӢеҖјз»ҷi
			if(group[j]!=group[i])
				in[group[j]]++;
		}
	}for(int i=1;i<=cnt;i++){
		if(in[i]==0){ //еӣ дёәе…ҘеәҰдёә0пјҢжүҖд»Ҙи®ҝй—®еҲ°иҝҷдёӘе…·зӣёеҪ“дәҺе°ҶиҝҷдёҖеқЁйғҪиҙӯд№°дәҶпјҢжҜ”д№°жңүе…ҘеәҰзҡ„зӮ№еҲ’з®—
			ans+=min_cost[i];
			if(min_cost[i]==0x3f3f3f3f){ //еҰӮжһңиҝҷдёӘеҢәй—ҙйғҪдёҚиғҪд№°е°ұдёҚиЎҢ
				cout<<"NO"<<endl;
				cout<<find(i)<<endl;
				return 0;
			}
		}
	}cout<<"YES"<<endl;
	cout<<ans<<endl;
	return 0;
}
