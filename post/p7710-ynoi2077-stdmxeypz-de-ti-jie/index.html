题目大意

把 aaa 子树中所有与 aaa 的距离模 xxx 等于 yyy 的节点权值加 zzz，并支持单点查询。

思路

根据经验，容易得到对于子树操作可以使用 dfs 序将一个子树转化为一个区间。假设节点 iii 的时间戳为 dfnidfn_idfni​ 且以 iii 为根的子树的大小为 sizeisize_isizei​，那么如果点 kkk 满足 dfnk∈[dfna,dfna+sizea]dfn_k\in[dfn_a,dfn_a+size_a]dfnk​∈[dfna​,dfna​+sizea​] 那么易得 kkk 为 aaa 的子树内一点。

在把 aaa 子树中所有与 aaa 的距离模 xxx 等于 yyy 的节点满足他们的距离之差为 yyy 的倍数。也就是说如果 kkk 会被修改，那么它必然满足 dfnk∈[dfna,dfna+sizea]dfn_k\in[dfn_a,dfn_a+size_a]dfnk​∈[dfna​,dfna​+sizea​] 且 (depa−depk)mod⁡x=y(dep_a-dep_k)\operatorname{mod}x=y(depa​−depk​)modx=y。

对于 ppp 我们可以直接暴力跳，因为 x>nx> \sqrt{n}x>n​ 那么即使整棵树是一个链即深度为 nnn 需要访问的点也不超过 n\sqrt{n}n​ 个，时间复杂度可以接受。对于查询，找到深度的时间为 O(1)O(1)O(1) 。

考虑继续使用上面的方法求解，发现暴力跳的时间复杂度 O(nx)O(\dfrac{n}{x})O(xn​) 可能会退化成为 O(n)O(n)O(n) 的不可以接受，所以考虑新的方法。

我们可以将 mdfx,(depa+y)mod⁡xmdf_{x,(dep_a+y)\operatorname{mod} x}mdfx,(depa​+y)modx​ 中 [dfna,dfna+sizea][dfn_a,dfn_a+size_a][dfna​,dfna​+sizea​] 全部加上 zzz，找到 mdfx,(depa+y)mod⁡xmdf_{x,(dep_a+y)\operatorname{mod} x}mdfx,(depa​+y)modx​ 的时间复杂度为 O(1)O(1)O(1)。

对于查询操作可以枚举模数 xxx，节点 aaa 的答案就是 ∑i=1⌊n⌋mdfi,depamod⁡i,dfna\sum_{i=1}^{\lfloor \sqrt{n}\rfloor} mdf_{i,dep_a\operatorname{mod} i,dfn_a}∑i=1⌊n​⌋​mdfi,depa​modi,dfna​​，找到 iii 和 depadep_adepa​ 时间复杂度的 O(n)O(\sqrt{n})O(n​)。这个式子很好理解，之所以值枚举到 n\sqrt{n}n​ 是因为所有第一维的修改都只是 xxx 而只有 x≤nx\le \sqrt{n}x≤n​ 的时候才会有修改，所以自然只需要遍历到 n\sqrt{n}n​ 就可以找到所有数了。

#include<iostream>
#include<vector>
#include<math.h>
using namespace std;
const int N=3e5+5,M=180,LEN=2000 ;
int n,f[N],dfn[N],siz[N],dep[N],cnt,mp[N],m,l[N],r[N],pos[N],q;
long long val[N],diff[M][N],mdf[M][305][305];
// mp[dfn[i]]=i 
vector<int> v[N];
void dfs(int x,int fa){
    siz[x]=1; 
	dep[x]=dep[fa]+1;
    dfn[x]=++cnt;
	mp[cnt]=x;
    for(int i:v[x]){
        if(i!=fa){
            dfs(i,x);
            siz[x]+=siz[i];
        }
    }
}
void pre(){
    m=n/LEN;
    for(int i=1;i<=m;i++){
        l[i]=(i-1)*LEN+1;
        r[i]=i*LEN;
    }
    if(r[m]<n){
        m++;
        l[m]=r[m-1]+1;
        r[m]=n;
    }
    for(int i=1;i<=m;i++){
        for(int j=l[i];j<=r[i];j++){
            pos[j]=i;
        }
    }
}
void update(int l,int r,int a,int b,int w){
    for (int i=l;i<=r;i++){
        if (dep[mp[i]]%a==b){
            val[i]+=w;
        }
    }
}
void change(int u,int a,int b,int w){
    int x=dfn[u],y=dfn[u]+siz[u]-1;
    int xx=pos[x],yy=pos[y];
    int now=(dep[u]+b)%a;
    if (xx==yy){ //散块 
        update(x,y,a,now,w);
        return;
    }
    update(x,r[xx],a,now,w);
    update(l[yy],y,a,now,w);
    if(a>300){ //第一种情况
        for(int i=dep[u]+b;i<=n;i+=a){
            diff[xx+1][i]+=w;
            diff[yy][i]-=w;
        }
    }
    else{ //第二种情况 
        for(int i=xx+1;i<yy;i++){
            mdf[i][a][now]+=w;
        }
    }
}
long long ask(int u){
    long long ans=val[dfn[u]];
    for (int i=1;i<=pos[dfn[u]];i++){
        ans+=diff[i][dep[u]];
    }
    for (int i=1;i<=300;i++){
        ans+=mdf[pos[dfn[u]]][i][dep[u]%i];
    }
    return ans;
}
signed main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cin>>n>>q;
    for(int i=2;i<=n;i++){
        cin>>f[i];
        v[f[i]].push_back(i);
    }
    dfs(1,0);
    pre();
    for(int i=1,op,a,x,y,z;i<=q;i++){
        cin>>op>>a;
        if(op==1){
            cin>>x>>y>>z;
            change(a,x,y,z);;
        }
        else{
            cout<<ask(a)<<'\n';
        }
    }
    return 0;
}
